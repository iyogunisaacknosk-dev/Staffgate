SQL — Run Once in Supabase

create table staff (
  id uuid primary key default gen_random_uuid(),
  name text not null,
  department text,
  email text,
  token text unique not null,
  unique_code text unique not null,
  closing_time time not null default '17:00',
  active boolean default true,
  created_at timestamptz default now()
);
create table exit_approvals (
  id uuid primary key default gen_random_uuid(),
  staff_id uuid references staff(id),
  reason text,
  used boolean default false,
  logged_at timestamptz default now()
);
create table access_log (
  id uuid primary key default gen_random_uuid(),
  staff_id uuid references staff(id),
  direction text,
  result text,
  via text,
  reason text,
  scanned_at timestamptz default now()
);
alter table staff enable row level security;
alter table exit_approvals enable row level security;
alter table access_log enable row level security;
create policy "open" on staff for all using (true) with check (true);
create policy "open" on exit_approvals for all using (true) with check (true);
create policy "open" on access_log for all using (true) with check (true);
